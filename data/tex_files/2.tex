    \draw[fill=gray!30] (2,1) rectangle ++(0.5,0.5);

    \coordinate (B) at ($(A)!1cm!90:(4,2)$);
    \coordinate (C) at ($(A)!1cm!-90:(4,0)$);
    \coordinate (D) at ($(A)!1cm!0:(4,0)$);
    
    % Draw vectors for forces
    \draw[-latex,red] (A) -- (B) node[midway,left] {Normal Force};
    \draw[-latex,blue] (A) -- (C) node[midway,right] {Weight};
    \draw[-latex,green] (A) -- (D) node[midway,above] {Friction};
