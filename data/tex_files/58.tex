    \draw (30:3.5cm) -- ++(90:\normal) node[midway,left]{$F_N$};
    \draw (30:3.5cm) -- ++(-90:\gravity) node[midway,right]{$F_g$};
    \draw (30:3.5cm) -- ++(210:\friction) node[midway,above]{$F_f$};
    \draw (30:3.5cm) -- ++(-60:\parallel) node[midway,below]{$F_{g\parallel}$};
\end{scope}
